 69    def task_export(self, id: int | str = None) -> list | None:
 70        """Get the raw task export of all tasks or a given task if the task id is set
 71
 72        Args:
 73            id (int | str, optional): The task identifier or the task uuid. Defaults to None.
 74
 75        Raises:
 76            TaskCommandNotFound: Could not find the task command - Is TaskWarrior installed?
 77            ErrorRunningTaskCommand: Unknown error running the task command
 78
 79        Returns:
 80            list | None: _description_
 81        """
 82        cmd = [self.get_task_command(), 'export']
 83        if id:
 84            cmd.insert(1, id)
 85        task_call = run(cmd, capture_output=True, text=True)
 86        if not task_call.returncode == 0:
 87            if task_call.returncode == 127:
 88                raise TaskCommandNotFound(task_call.stderr)
 89            raise ErrorRunningTaskCommand(task_call.stderr)
 90
 91        data = ast.literal_eval(task_call.stdout)
 92        if len(data) >= 1:
 93            return data
 94        return None

 96    def get_all_tasks(self, include_deleted: bool = False) -> list[Task]:
 97        """Get all TaskWarrior tasks

116                task_id = task_data.get('id')
117                if not task_id == 0:
118                    self.working_set.append(Task(task_data))

120                all_tasks.append(Task(task_data))
121
122        return all_tasks
